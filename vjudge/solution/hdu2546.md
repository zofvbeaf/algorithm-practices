#[hdu2546 饭卡](http://acm.hdu.edu.cn/showproblem.php?pid=2546)
#SOLUTION
+ 找出最贵的菜花5元买之，转化为背包容量为m-5的01背包问题
+ 复杂度为O(n)
#AC CODE
```c++
#include <cstdio>
#include <algorithm>
#include <string.h>
using namespace std;
int main(){
    int n, m, a[2000], maxa, ans;
    int dp[2000];
    while(~scanf("%d", &n) && n) {
        memset(dp, 0, sizeof(dp));
        maxa = ans = 0;
        for( int i = 0; i < n; i++){
            scanf("%d", &a[i]);
            if(a[maxa] < a[i]) maxa = i; 
        }
        scanf("%d", &m);
        if(m < 5){   //小于5元不能买任何东西
            printf("%d\n", m);
            continue;
        } 
        ans = a[maxa], a[maxa] = m + 1; //找出最贵的菜，花5元买之
        m -= 5;   //背包总容量变为 m-5 , 转化为01背包问题
        for( int i = 0; i < n; i++)
            for( int j = m; j >= a[i]; j--)
                dp[j] = max(dp[j], dp[j-a[i]] + a[i]);
        printf("%d\n", m + 5 - (ans + dp[m]));
    }
    return 0;
}
```
