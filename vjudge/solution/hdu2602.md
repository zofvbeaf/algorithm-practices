#AC CODE
```c++
#include <cstdio>
#include <algorithm>
#include <string.h>
using namespace std;
int main(){
    int t, n, v, c[2000], w[2000];
    int dp[2000];
    scanf("%d", &t);
    while(t--) {
        scanf("%d%d", &n, &v);
        memset(dp, 0, sizeof(dp));
        for( int i = 0; i < n; i++)
            scanf("%d", &w[i]);
        for( int i = 0; i < n; i++)
            scanf("%d", &c[i]);
        for( int i = 0; i < n; i++)
            for( int j = v; j >= c[i]; j--)
                dp[j] = max(dp[j], dp[j-c[i]] + w[i]);
        printf("%d\n", dp[v]);      
    }
    return 0;
}
```
