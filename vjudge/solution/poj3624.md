#[poj3624 Charm Bracelet](http://poj.org/problem?id=3624)
#SOLUTION
+ 超级标准的01背包
#AC CODE
```c++
#include <cstdio>
#include <algorithm>
#include <string.h>
using namespace std;
int main(){
    int n, v, c[4000], w[4000];
    int dp[13880];
    while(~scanf("%d%d", &n, &v)) {
        memset(dp, 0, sizeof(dp));
        for( int i = 0; i < n; i++)
            scanf("%d%d", &c[i], &w[i]);
        for( int i = 0; i < n; i++)
            for( int j = v; j >= c[i]; j--)
                dp[j] = max(dp[j], dp[j-c[i]] + w[i]);
        printf("%d\n", dp[v]);      
    }
    return 0;
}
```
